  <script>
    const STORAGE_KEY = 'nrn_ai_chat_v1';

    const messagesEl = document.getElementById('messages');
    const typingEl = document.getElementById('typing');
    const inputEl = document.getElementById('input');
    const sendBtn = document.getElementById('sendBtn');
    const clearBtn = document.getElementById('clearBtn');
    const exportBtn = document.getElementById('exportBtn');

    /** @type {{role:'me'|'ai', text:string, ts:number}[]} */
    let chat = [];

    function nowTime(){
      const d = new Date();
      return d.toLocaleString();
    }

    function save(){
      localStorage.setItem(STORAGE_KEY, JSON.stringify(chat));
    }

    function load(){
      try{
        const raw = localStorage.getItem(STORAGE_KEY);
        chat = raw ? JSON.parse(raw) : [];
      }catch{
        chat = [];
      }
    }

    function addMessage(role, text){
      chat.push({ role, text, ts: Date.now() });
      save();
      render();
    }

    function render(){
      messagesEl.innerHTML = '';
      for(const m of chat){
        const div = document.createElement('div');
        div.className = 'msg ' + (m.role === 'me' ? 'me' : 'ai');
        div.textContent = m.text;
        messagesEl.appendChild(div);
      }
      messagesEl.scrollTop = messagesEl.scrollHeight;
    }

    function setTyping(on){
      typingEl.textContent = on ? 'NRN AI is typing…' : '';
    }

    function demoBrain(userText){
      const t = (userText || '').trim().toLowerCase();

      if(t === 'help'){
        return [
          "Commands:",
          "- help: show commands",
          "- apps: ideas for Windows 7 apps",
          "- about: what this assistant is",
          "- time: current device time",
          "",
          "Tip: Ask: 'suggest new apps' or 'how to add mini chrome app'."
        ].join('\n');
      }

      if(t === 'about'){
        return "I’m NRN AI (demo). I can run fully in your HTML portfolio without any API keys. If you want real Gemini responses, add a secure backend proxy so your key stays hidden.";
      }

      if(t === 'time'){
        return "Your current device time is: " + nowTime();
      }

      if(t === 'apps' || t.includes('suggest') || t.includes('apps')){
        return [
          "More Windows-style apps you can add (HTML/JS):",
          "1) Run dialog (Win+R) to launch apps",
          "2) Recycle Bin with restore/empty",
          "3) Task Manager listing open windows",
          "4) File Explorer with folders + file types",
          "5) Windows Media Player mini",
          "",
          "If you tell me your app folder names, I can generate a Start Menu + Run command map."
        ].join('\n');
      }

      if(t.includes('chrome') || t.includes('browser')){
        return [
          "Mini browser tips:",
          "- Use an iframe for internal pages.",
          "- Many external sites block iframe embedding (normal).",
          "- Provide 'Open in real browser' button using window.open(url).",
          "",
          "Your chrome.html I gave you is built this way."
        ].join('\n');
      }

      if(t.includes('gemini') || t.includes('api key')){
        return [
          "To use Gemini safely:",
          "- DO NOT put the key in frontend JS (it will leak).",
          "- Use a free backend proxy (Cloudflare Worker / Vercel Function).",
          "- Your HTML calls your proxy; proxy calls Gemini.",
          "",
          "Reply: 'I can use Cloudflare Worker' and I’ll send exact code."
        ].join('\n');
      }

      return "I can help. Try 'apps' for ideas, or tell me what you want to add next (browser, control panel, explorer, task manager, etc.).";
    }

    async function send(){
      const text = inputEl.value.trim();
      if(!text) return;

      inputEl.value = '';
      addMessage('me', text);

      setTyping(true);
      // Simulate thinking
      await new Promise(r => setTimeout(r, 450));

      const reply = demoBrain(text);
      addMessage('ai', reply);
      setTyping(false);
    }

    sendBtn.addEventListener('click', send);
    inputEl.addEventListener('keydown', (e) => {
      if(e.key === 'Enter' && !e.shiftKey){
        e.preventDefault();
        send();
      }
    });

    clearBtn.addEventListener('click', () => {
      if(confirm('Clear chat history?')){
        chat = [];
        save();
        render();
      }
    });

    exportBtn.addEventListener('click', () => {
      const blob = new Blob([JSON.stringify(chat, null, 2)], {type:'application/json'});
      const a = document.createElement('a');
      a.href = URL.createObjectURL(blob);
      a.download = 'nrn-ai-chat.json';
      a.click();
      URL.revokeObjectURL(a.href);
    });

    // init
    load();
    if(!chat.length){
      addMessage('ai', "Hi! I’m NRN AI (demo). Type 'help' to see commands.");
    } else {
      render();
    }
  </script>
